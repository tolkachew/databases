drop table if exists student;
drop table if exists university;

create table university
(
  id integer primary key autoincrement,
  name varchar(160)
);

create table student
(
  id integer primary key autoincrement,
  surname varchar(60),
  name varchar(60),
  univ_id integer,
  foreign key (univ_id) references university (id)
  ON DELETE CASCADE
);
